# 防抖：将一段时间内的多次高频操作优化为一次操作

# 通常使用场景: 用户输入，在输入完成后进行校验

//节流
const debounce = (fn, wait) => {
    let timer;
    return function(...args){
        const context = this
        if(timer){
            clearTimeout(timer)
        }else{
            timer = setTimeout(() => {
                fn.apply(context, args)
            }, wait)
        }
        
    }
}

//防抖
const throttle = (fn, wait) => {
    let timer;
    return function(...args){
        const context = this
        if(!timer){
            timer = setTimeout(() => {
                fn.apply(context, args)
                timer = null
            }, wait)
        }
    }
}
