# 定义自己的 new 函数
// 不依赖this 所以使用箭头函数定义更简洁  同时避免函数提升
const newFun = (fn, ...params) => {
    
    // 定义一个空对象 
    const p = Object.create(fn.prototype);

    //将p的原型指向传入函数的原型    
    Object.setPrototypeOf(p, fn.prototype);

    //调用call函数将fn的执行环境设置为p
    const res = fn.call(p, ...params);

    //判断函数的返回值是否为对象类型 如果是则之间返回，如果不是则返回p
    return typeof res === 'object' ? res : p;
}

const newFun = (fn, ...params) => {
    const res = fn.call(fn.prototype, ...params)
    
    return typeof res === 'object' ? res : fn.prototype;
}
